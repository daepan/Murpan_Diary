## DP
- Top-down (Memoization) 구현
```C++
#include <iostream>
#include <vector>
using namespace std;

int fib(int n, vector<int>& dp) {
    if (n <= 1) return n;
    if (dp[n] != -1) return dp[n];
    return dp[n] = fib(n-1, dp) + fib(n-2, dp);
}

int main() {
    int n = 10;
    vector<int> dp(n+1, -1);
    cout << fib(n, dp) << endl;
    return 0;
}
```
- Bottom-up
```C++
#include <iostream>
#include <vector>
using namespace std;

int fib(int n) {
    vector<int> dp(n+1);
    dp[0] = 0;
    dp[1] = 1;
    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i-1] + dp[i-2];
    }
    return dp[n];
}

int main() {
    int n = 10;
    cout << fib(n) << endl;  // 10번째 피보나치 수 출력
    return 0;
}
```

## 다익스트라
```C++
#include <iostream>
#include <vector>
#include <queue>
#include <limits>
using namespace std;

const int INF = numeric_limits<int>::max();

void dijkstra(int start, vector<vector<pair<int, int>>>& graph, vector<int>& dist) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq;
    dist[start] = 0;
    pq.push({0, start});

    while (!pq.empty()) {
        int distance = pq.top().first;
        int node = pq.top().second;
        pq.pop();

        if (dist[node] < distance) continue;

        for (auto& edge : graph[node]) {
            int next = edge.first;
            int nextDist = distance + edge.second;

            if (nextDist < dist[next]) {
                dist[next] = nextDist;
                pq.push({nextDist, next});
            }
        }
    }
}

int main() {
    int V = 5;  // 노드 개수
    vector<vector<pair<int, int>>> graph(V);
    
    // 예시로 그래프 생성 (노드 0과 1 사이에 가중치 2)
    graph[0].push_back({1, 2});
    graph[1].push_back({2, 3});
    graph[2].push_back({3, 1});
    graph[3].push_back({4, 5});
    
    vector<int> dist(V, INF);
    dijkstra(0, graph, dist);  // 0번 노드에서 시작

    for (int i = 0; i < V; i++) {
        cout << "Distance to node " << i << ": " << dist[i] << endl;
    }

    return 0;
}
```

## 세그먼트 트리
특정 구간의 값을 빠르게 업데이트하거나 쿼리를 처리
```C++
#include <iostream>
#include <vector>
using namespace std;

vector<int> segtree;
vector<int> arr;

// 세그먼트 트리 빌드
void build(int node, int start, int end) {
    if (start == end) {
        segtree[node] = arr[start];
    } else {
        int mid = (start + end) / 2;
        build(2 * node, start, mid);
        build(2 * node + 1, mid + 1, end);
        segtree[node] = segtree[2 * node] + segtree[2 * node + 1];
    }
}

// 구간 합 쿼리
int query(int node, int start, int end, int l, int r) {
    if (r < start || end < l) {
        return 0;  // 구간이 겹치지 않는 경우
    }
    if (l <= start && end <= r) {
        return segtree[node];  // 완전히 포함되는 경우
    }
    int mid = (start + end) / 2;
    int left_sum = query(2 * node, start, mid, l, r);
    int right_sum = query(2 * node + 1, mid + 1, end, l, r);
    return left_sum + right_sum;
}

// 업데이트
void update(int node, int start, int end, int idx, int val) {
    if (start == end) {
        arr[idx] = val;
        segtree[node] = val;
    } else {
        int mid = (start + end) / 2;
        if (start <= idx && idx <= mid) {
            update(2 * node, start, mid, idx, val);
        } else {
            update(2 * node + 1, mid + 1, end, idx, val);
        }
        segtree[node] = segtree[2 * node] + segtree[2 * node + 1];
    }
}

int main() {
    arr = {1, 2, 3, 4, 5};  // 원본 배열
    int n = arr.size();
    segtree.resize(4 * n);
    
    build(1, 0, n-1);  // 세그먼트 트리 빌드

    cout << query(1, 0, n-1, 1, 3) << endl;  // 구간 합 [1, 3]
    update(1, 0, n-1, 2, 10);  // 인덱스 2의 값을 10으로 업데이트
    cout << query(1, 0, n-1, 1, 3) << endl;  // 구간 합 [1, 3]

    return 0;
}
```
## 최소공통조상 LCA
그래프 문제나 트리 기반 문제
```C++
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

vector<vector<int>> up;  // Sparse table
vector<int> depth;

void dfs(int node, int parent, const vector<vector<int>>& tree) {
    up[node][0] = parent;
    for (int i = 1; i < up[0].size(); i++) {
        if (up[node][i-1] != -1)
            up[node][i] = up[up[node][i-1]][i-1];
    }
    for (int neighbor : tree[node]) {
        if (neighbor != parent) {
            depth[neighbor] = depth[node] + 1;
            dfs(neighbor, node, tree);
        }
    }
}

int lca(int u, int v) {
    if (depth[u] < depth[v]) swap(u, v);

    int log = up[0].size();
    for (int i = log - 1; i >= 0; i--) {
        if (depth[u] - (1 << i) >= depth[v]) {
            u = up[u][i];
        }
    }

    if (u == v) return u;

    for (int i = log - 1; i >= 0; i--) {
        if (up[u][i] != up[v][i]) {
            u = up[u][i];
            v = up[v][i];
        }
    }

    return up[u][0];
}

int main() {
    int n = 5;
    vector<vector<int>> tree(n);
    tree[0] = {1, 2};  // 트리 그래프 설정
    tree[1] = {3, 4};

    int log = ceil(log2(n));
    up.assign(n, vector<int>(log+1, -1));
    depth.assign(n, 0);

    dfs(0, -1, tree);  // 루트 노드 0에서 시작하는 DFS

    cout << "LCA of 3 and 4: " << lca(3, 4) << endl;

    return 0;
}
```
## 비트 마스크
집합을 비트로 표현하여 모든 부분집합을 탐색하는 문제나, 최적화 문제
```C++
#include <iostream>
using namespace std;

int main() {
    int n = 3;  // 집합 크기 (원소가 3개)
    
    // 2^n개의 부분집합 탐색
    for (int bit = 0; bit < (1 << n); bit++) {
        cout << "{ ";
        for (int i = 0; i < n; i++) {
            if (bit & (1 << i)) {
                cout << i << " ";
            }
        }
        cout << "}" << endl;
    }

    return 0;
}
```
## 트라이
접두사 검색(Prefix Search) 문제나, 사전 형태의 문자열 저장 및 탐색 문제
```C++
#include <iostream>
#include <unordered_map>
using namespace std;

struct TrieNode {
    unordered_map<char, TrieNode*> children;
    bool isEndOfWord;
    
    TrieNode() : isEndOfWord(false) {}
};

class Trie {
public:
    TrieNode* root;
    
    Trie() {
        root = new TrieNode();
    }
    
    // 문자열 삽입
    void insert(string word) {
        TrieNode* node = root;
        for (char ch : word) {
            if (!node->children.count(ch)) {
                node->children[ch] = new TrieNode();
            }
            node = node->children[ch];
        }
        node->isEndOfWord = true;
    }
    
    // 문자열 탐색
    bool search(string word) {
        TrieNode* node = root;
        for (char ch : word) {
            if (!node->children.count(ch)) {
                return false;
            }
            node = node->children[ch];
        }
        return node->isEndOfWord;
    }
    
    // 접두사 탐색
    bool startsWith(string prefix) {
        TrieNode* node = root;
        for (char ch : prefix) {
            if (!node->children.count(ch)) {
                return false;
            }
            node = node->children[ch];
        }
        return true;
    }
};

int main() {
    Trie trie;
    trie.insert("hello");
    trie.insert("helium");
    
    cout << trie.search("hello") << endl;  // true
    cout << trie.startsWith("he") << endl;  // true
    cout << trie.search("hero") << endl;  // false
    
    return 0;
}
```
## 디닉 알고리즘
```C++
#include <iostream>
#include <vector>
#include <queue>
#include <climits>
using namespace std;

struct Edge {
    int to, rev;
    int flow, cap;
};

class Dinic {
    vector<vector<Edge>> adj;
    vector<int> level;
    vector<int> start;
    int n;

public:
    Dinic(int n) : n(n), adj(n), level(n), start(n) {}

    void addEdge(int u, int v, int cap) {
        adj[u].push_back({v, (int)adj[v].size(), 0, cap});
        adj[v].push_back({u, (int)adj[u].size() - 1, 0, 0});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        level[s] = 0;
        queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (Edge& e : adj[u]) {
                if (level[e.to] == -1 && e.flow < e.cap) {
                    level[e.to] = level[u] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] != -1;
    }

    int dfs(int u, int t, int flow) {
        if (u == t) return flow;
        for (int& i = start[u]; i < adj[u].size(); i++) {
            Edge& e = adj[u][i];
            if (level[e.to] == level[u] + 1 && e.flow < e.cap) {
                int curr_flow = min(flow, e.cap - e.flow);
                int temp_flow = dfs(e.to, t, curr_flow);
                if (temp_flow > 0) {
                    e.flow += temp_flow;
                    adj[e.to][e.rev].flow -= temp_flow;
                    return temp_flow;
                }
            }
        }
        return 0;
    }

    int maxFlow(int s, int t) {
        int total = 0;
        while (bfs(s, t)) {
            fill(start.begin(), start.end(), 0);
            while (int flow = dfs(s, t, INT_MAX)) {
                total += flow;
            }
        }
        return total;
    }
};

int main() {
    Dinic dinic(4);
    dinic.addEdge(0, 1, 3);
    dinic.addEdge(0, 2, 2);
    dinic.addEdge(1, 2, 1);
    dinic.addEdge(1, 3, 2);
    dinic.addEdge(2, 3, 3);

    cout << "Maximum Flow: " << dinic.maxFlow(0, 3) << endl;

    return 0;
}
```

## DP + 분할정복
```C++
#include <iostream>
#include <vector>
#include <limits.h>
using namespace std;

int matrixChainMultiplication(vector<int>& dims) {
    int n = dims.size() - 1;
    vector<vector<int>> dp(n, vector<int>(n, 0));

    for (int len = 2; len <= n; len++) {
        for (int i = 0; i < n - len + 1; i++) {
            int j = i + len - 1;
            dp[i][j] = INT_MAX;
            for (int k = i; k < j; k++) {
                int cost = dp[i][k] + dp[k+1][j] + dims[i] * dims[k+1] * dims[j+1];
                dp[i][j] = min(dp[i][j], cost);
            }
        }
    }
    return dp[0][n-1];
}

int main() {
    vector<int> dims = {1, 2, 3, 4};
    cout << "Minimum multiplication cost: " << matrixChainMultiplication(dims) << endl;
    return 0;
}
```

## 미니맥스 알고리즘
```C++
#include <iostream>
#include <vector>
using namespace std;

const int INF = 1000;

int minimax(vector<int>& values, int depth, bool isMax) {
    if (depth == values.size()) return 0;
    if (isMax) {
        int best = -INF;
        for (int i = 0; i < values.size(); i++) {
            best = max(best, minimax(values, depth + 1, !isMax));
        }
        return best;
    } else {
        int best = INF;
        for (int i = 0; i < values.size(); i++) {
            best = min(best, minimax(values, depth + 1, !isMax));
        }
        return best;
    }
}

int main() {
    vector<int> values = {3, 5, 2, 9, 12, 5};
    cout << "Optimal value: " << minimax(values, 0, true) << endl;
    return 0;
}
```
## 허프만 코딩
```C++
#include <iostream>
#include <queue>
#include <unordered_map>
using namespace std;

struct Node {
    char ch;
    int freq;
    Node *left, *right;
    Node(char ch, int freq) : ch(ch), freq(freq), left(nullptr), right(nullptr) {}
};

struct Compare {
    bool operator()(Node* l, Node* r) {
        return l->freq > r->freq;
    }
};

void printHuffmanCodes(Node* root, string code) {
    if (!root) return;
    if (root->ch != '#') cout << root->ch << ": " << code << endl;
    printHuffmanCodes(root->left, code + "0");
    printHuffmanCodes(root->right, code + "1");
}

void huffmanCoding(string text) {
    unordered_map<char, int> freq;
    for (char ch : text) freq[ch]++;

    priority_queue<Node*, vector<Node*>, Compare> pq;
    for (auto& pair : freq) {
        pq.push(new Node(pair.first, pair.second));
    }

    while (pq.size() > 1) {
        Node* left = pq.top(); pq.pop();
        Node* right = pq.top(); pq.pop();
        Node* combined = new Node('#', left->freq + right->freq);
        combined->left = left;
        combined->right = right;
        pq.push(combined);
    }

    Node* root = pq.top();
    printHuffmanCodes(root, "");
}

int main() {
    string text = "this is an example of huffman coding";
    huffmanCoding(text);
    return 0;
}
```
