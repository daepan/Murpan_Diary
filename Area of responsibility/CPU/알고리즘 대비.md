## DP
- Top-down (Memoization) 구현
```C++
#include <iostream>
#include <vector>
using namespace std;

int fib(int n, vector<int>& dp) {
    if (n <= 1) return n;
    if (dp[n] != -1) return dp[n];
    return dp[n] = fib(n-1, dp) + fib(n-2, dp);
}

int main() {
    int n = 10;
    vector<int> dp(n+1, -1);
    cout << fib(n, dp) << endl;
    return 0;
}
```
- Bottom-up
```C++
#include <iostream>
#include <vector>
using namespace std;

int fib(int n) {
    vector<int> dp(n+1);
    dp[0] = 0;
    dp[1] = 1;
    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i-1] + dp[i-2];
    }
    return dp[n];
}

int main() {
    int n = 10;
    cout << fib(n) << endl;  // 10번째 피보나치 수 출력
    return 0;
}
```

## 다익스트라
```C++
#include <iostream>
#include <vector>
#include <queue>
#include <limits>
using namespace std;

const int INF = numeric_limits<int>::max();

void dijkstra(int start, vector<vector<pair<int, int>>>& graph, vector<int>& dist) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq;
    dist[start] = 0;
    pq.push({0, start});

    while (!pq.empty()) {
        int distance = pq.top().first;
        int node = pq.top().second;
        pq.pop();

        if (dist[node] < distance) continue;

        for (auto& edge : graph[node]) {
            int next = edge.first;
            int nextDist = distance + edge.second;

            if (nextDist < dist[next]) {
                dist[next] = nextDist;
                pq.push({nextDist, next});
            }
        }
    }
}

int main() {
    int V = 5;  // 노드 개수
    vector<vector<pair<int, int>>> graph(V);
    
    // 예시로 그래프 생성 (노드 0과 1 사이에 가중치 2)
    graph[0].push_back({1, 2});
    graph[1].push_back({2, 3});
    graph[2].push_back({3, 1});
    graph[3].push_back({4, 5});
    
    vector<int> dist(V, INF);
    dijkstra(0, graph, dist);  // 0번 노드에서 시작

    for (int i = 0; i < V; i++) {
        cout << "Distance to node " << i << ": " << dist[i] << endl;
    }

    return 0;
}
```

## 세그먼트 트리
```C++
#include <iostream>
#include <vector>
using namespace std;

vector<int> segtree;
vector<int> arr;

// 세그먼트 트리 빌드
void build(int node, int start, int end) {
    if (start == end) {
        segtree[node] = arr[start];
    } else {
        int mid = (start + end) / 2;
        build(2 * node, start, mid);
        build(2 * node + 1, mid + 1, end);
        segtree[node] = segtree[2 * node] + segtree[2 * node + 1];
    }
}

// 구간 합 쿼리
int query(int node, int start, int end, int l, int r) {
    if (r < start || end < l) {
        return 0;  // 구간이 겹치지 않는 경우
    }
    if (l <= start && end <= r) {
        return segtree[node];  // 완전히 포함되는 경우
    }
    int mid = (start + end) / 2;
    int left_sum = query(2 * node, start, mid, l, r);
    int right_sum = query(2 * node + 1, mid + 1, end, l, r);
    return left_sum + right_sum;
}

// 업데이트
void update(int node, int start, int end, int idx, int val) {
    if (start == end) {
        arr[idx] = val;
        segtree[node] = val;
    } else {
        int mid = (start + end) / 2;
        if (start <= idx && idx <= mid) {
            update(2 * node, start, mid, idx, val);
        } else {
            update(2 * node + 1, mid + 1, end, idx, val);
        }
        segtree[node] = segtree[2 * node] + segtree[2 * node + 1];
    }
}

int main() {
    arr = {1, 2, 3, 4, 5};  // 원본 배열
    int n = arr.size();
    segtree.resize(4 * n);
    
    build(1, 0, n-1);  // 세그먼트 트리 빌드

    cout << query(1, 0, n-1, 1, 3) << endl;  // 구간 합 [1, 3]
    update(1, 0, n-1, 2, 10);  // 인덱스 2의 값을 10으로 업데이트
    cout << query(1, 0, n-1, 1, 3) << endl;  // 구간 합 [1, 3]

    return 0;
}
```
## 최소공통조상 LCA
```C++
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

vector<vector<int>> up;  // Sparse table
vector<int> depth;

void dfs(int node, int parent, const vector<vector<int>>& tree) {
    up[node][0] = parent;
    for (int i = 1; i < up[0].size(); i++) {
        if (up[node][i-1] != -1)
            up[node][i] = up[up[node][i-1]][i-1];
    }
    for (int neighbor : tree[node]) {
        if (neighbor != parent) {
            depth[neighbor] = depth[node] + 1;
            dfs(neighbor, node, tree);
        }
    }
}

int lca(int u, int v) {
    if (depth[u] < depth[v]) swap(u, v);

    int log = up[0].size();
    for (int i = log - 1; i >= 0; i--) {
        if (depth[u] - (1 << i) >= depth[v]) {
            u = up[u][i];
        }
    }

    if (u == v) return u;

    for (int i = log - 1; i >= 0; i--) {
        if (up[u][i] != up[v][i]) {
            u = up[u][i];
            v = up[v][i];
        }
    }

    return up[u][0];
}

int main() {
    int n = 5;
    vector<vector<int>> tree(n);
    tree[0] = {1, 2};  // 트리 그래프 설정
    tree[1] = {3, 4};

    int log = ceil(log2(n));
    up.assign(n, vector<int>(log+1, -1));
    depth.assign(n, 0);

    dfs(0, -1, tree);  // 루트 노드 0에서 시작하는 DFS

    cout << "LCA of 3 and 4: " << lca(3, 4) << endl;

    return 0;
}
```
