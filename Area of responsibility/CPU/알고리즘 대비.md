## DP
- Top-down (Memoization) 구현
```C++
#include <iostream>
#include <vector>
using namespace std;

int fib(int n, vector<int>& dp) {
    if (n <= 1) return n;
    if (dp[n] != -1) return dp[n];
    return dp[n] = fib(n-1, dp) + fib(n-2, dp);
}

int main() {
    int n = 10;
    vector<int> dp(n+1, -1);
    cout << fib(n, dp) << endl;
    return 0;
}
```
- Bottom-up
```C++
#include <iostream>
#include <vector>
using namespace std;

int fib(int n) {
    vector<int> dp(n+1);
    dp[0] = 0;
    dp[1] = 1;
    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i-1] + dp[i-2];
    }
    return dp[n];
}

int main() {
    int n = 10;
    cout << fib(n) << endl;  // 10번째 피보나치 수 출력
    return 0;
}
```

## 다익스트라
```C++
#include <iostream>
#include <vector>
#include <queue>
#include <limits>
using namespace std;

const int INF = numeric_limits<int>::max();

void dijkstra(int start, vector<vector<pair<int, int>>>& graph, vector<int>& dist) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq;
    dist[start] = 0;
    pq.push({0, start});

    while (!pq.empty()) {
        int distance = pq.top().first;
        int node = pq.top().second;
        pq.pop();

        if (dist[node] < distance) continue;

        for (auto& edge : graph[node]) {
            int next = edge.first;
            int nextDist = distance + edge.second;

            if (nextDist < dist[next]) {
                dist[next] = nextDist;
                pq.push({nextDist, next});
            }
        }
    }
}

int main() {
    int V = 5;  // 노드 개수
    vector<vector<pair<int, int>>> graph(V);
    
    // 예시로 그래프 생성 (노드 0과 1 사이에 가중치 2)
    graph[0].push_back({1, 2});
    graph[1].push_back({2, 3});
    graph[2].push_back({3, 1});
    graph[3].push_back({4, 5});
    
    vector<int> dist(V, INF);
    dijkstra(0, graph, dist);  // 0번 노드에서 시작

    for (int i = 0; i < V; i++) {
        cout << "Distance to node " << i << ": " << dist[i] << endl;
    }

    return 0;
}
```

## 세그먼트 트리
특정 구간의 값을 빠르게 업데이트하거나 쿼리를 처리
```C++
#include <iostream>
#include <vector>
using namespace std;

vector<int> segtree;
vector<int> arr;

// 세그먼트 트리 빌드
void build(int node, int start, int end) {
    if (start == end) {
        segtree[node] = arr[start];
    } else {
        int mid = (start + end) / 2;
        build(2 * node, start, mid);
        build(2 * node + 1, mid + 1, end);
        segtree[node] = segtree[2 * node] + segtree[2 * node + 1];
    }
}

// 구간 합 쿼리
int query(int node, int start, int end, int l, int r) {
    if (r < start || end < l) {
        return 0;  // 구간이 겹치지 않는 경우
    }
    if (l <= start && end <= r) {
        return segtree[node];  // 완전히 포함되는 경우
    }
    int mid = (start + end) / 2;
    int left_sum = query(2 * node, start, mid, l, r);
    int right_sum = query(2 * node + 1, mid + 1, end, l, r);
    return left_sum + right_sum;
}

// 업데이트
void update(int node, int start, int end, int idx, int val) {
    if (start == end) {
        arr[idx] = val;
        segtree[node] = val;
    } else {
        int mid = (start + end) / 2;
        if (start <= idx && idx <= mid) {
            update(2 * node, start, mid, idx, val);
        } else {
            update(2 * node + 1, mid + 1, end, idx, val);
        }
        segtree[node] = segtree[2 * node] + segtree[2 * node + 1];
    }
}

int main() {
    arr = {1, 2, 3, 4, 5};  // 원본 배열
    int n = arr.size();
    segtree.resize(4 * n);
    
    build(1, 0, n-1);  // 세그먼트 트리 빌드

    cout << query(1, 0, n-1, 1, 3) << endl;  // 구간 합 [1, 3]
    update(1, 0, n-1, 2, 10);  // 인덱스 2의 값을 10으로 업데이트
    cout << query(1, 0, n-1, 1, 3) << endl;  // 구간 합 [1, 3]

    return 0;
}
```
## 트라이
접두사 검색(Prefix Search) 문제나, 사전 형태의 문자열 저장 및 탐색 문제
```C++
#include <iostream>
#include <unordered_map>
using namespace std;

struct TrieNode {
    unordered_map<char, TrieNode*> children;
    bool isEndOfWord;
    
    TrieNode() : isEndOfWord(false) {}
};

class Trie {
public:
    TrieNode* root;
    
    Trie() {
        root = new TrieNode();
    }
    
    // 문자열 삽입
    void insert(string word) {
        TrieNode* node = root;
        for (char ch : word) {
            if (!node->children.count(ch)) {
                node->children[ch] = new TrieNode();
            }
            node = node->children[ch];
        }
        node->isEndOfWord = true;
    }
    
    // 문자열 탐색
    bool search(string word) {
        TrieNode* node = root;
        for (char ch : word) {
            if (!node->children.count(ch)) {
                return false;
            }
            node = node->children[ch];
        }
        return node->isEndOfWord;
    }
    
    // 접두사 탐색
    bool startsWith(string prefix) {
        TrieNode* node = root;
        for (char ch : prefix) {
            if (!node->children.count(ch)) {
                return false;
            }
            node = node->children[ch];
        }
        return true;
    }
};

int main() {
    Trie trie;
    trie.insert("hello");
    trie.insert("helium");
    
    cout << trie.search("hello") << endl;  // true
    cout << trie.startsWith("he") << endl;  // true
    cout << trie.search("hero") << endl;  // false
    
    return 0;
}
```
## DP + 분할정복
```C++
#include <iostream>
#include <vector>
#include <limits.h>
using namespace std;

int matrixChainMultiplication(vector<int>& dims) {
    int n = dims.size() - 1;
    vector<vector<int>> dp(n, vector<int>(n, 0));

    for (int len = 2; len <= n; len++) {
        for (int i = 0; i < n - len + 1; i++) {
            int j = i + len - 1;
            dp[i][j] = INT_MAX;
            for (int k = i; k < j; k++) {
                int cost = dp[i][k] + dp[k+1][j] + dims[i] * dims[k+1] * dims[j+1];
                dp[i][j] = min(dp[i][j], cost);
            }
        }
    }
    return dp[0][n-1];
}

int main() {
    vector<int> dims = {1, 2, 3, 4};
    cout << "Minimum multiplication cost: " << matrixChainMultiplication(dims) << endl;
    return 0;
}
```

## 미니맥스 알고리즘
```C++
#include <iostream>
#include <vector>
using namespace std;

const int INF = 1000;

int minimax(vector<int>& values, int depth, bool isMax) {
    if (depth == values.size()) return 0;
    if (isMax) {
        int best = -INF;
        for (int i = 0; i < values.size(); i++) {
            best = max(best, minimax(values, depth + 1, !isMax));
        }
        return best;
    } else {
        int best = INF;
        for (int i = 0; i < values.size(); i++) {
            best = min(best, minimax(values, depth + 1, !isMax));
        }
        return best;
    }
}

int main() {
    vector<int> values = {3, 5, 2, 9, 12, 5};
    cout << "Optimal value: " << minimax(values, 0, true) << endl;
    return 0;
}
```
## 허프만 코딩
```C++
#include <iostream>
#include <queue>
#include <unordered_map>
using namespace std;

struct Node {
    char ch;
    int freq;
    Node *left, *right;
    Node(char ch, int freq) : ch(ch), freq(freq), left(nullptr), right(nullptr) {}
};

struct Compare {
    bool operator()(Node* l, Node* r) {
        return l->freq > r->freq;
    }
};

void printHuffmanCodes(Node* root, string code) {
    if (!root) return;
    if (root->ch != '#') cout << root->ch << ": " << code << endl;
    printHuffmanCodes(root->left, code + "0");
    printHuffmanCodes(root->right, code + "1");
}

void huffmanCoding(string text) {
    unordered_map<char, int> freq;
    for (char ch : text) freq[ch]++;

    priority_queue<Node*, vector<Node*>, Compare> pq;
    for (auto& pair : freq) {
        pq.push(new Node(pair.first, pair.second));
    }

    while (pq.size() > 1) {
        Node* left = pq.top(); pq.pop();
        Node* right = pq.top(); pq.pop();
        Node* combined = new Node('#', left->freq + right->freq);
        combined->left = left;
        combined->right = right;
        pq.push(combined);
    }

    Node* root = pq.top();
    printHuffmanCodes(root, "");
}

int main() {
    string text = "this is an example of huffman coding";
    huffmanCoding(text);
    return 0;
}
```
