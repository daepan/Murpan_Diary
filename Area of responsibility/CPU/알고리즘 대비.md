## DP
- Top-down (Memoization) 구현
```C++
#include <iostream>
#include <vector>
using namespace std;

int fib(int n, vector<int>& dp) {
    if (n <= 1) return n;
    if (dp[n] != -1) return dp[n];
    return dp[n] = fib(n-1, dp) + fib(n-2, dp);
}

int main() {
    int n = 10;
    vector<int> dp(n+1, -1);
    cout << fib(n, dp) << endl;
    return 0;
}
```
- Bottom-up
```C++
#include <iostream>
#include <vector>
using namespace std;

int fib(int n) {
    vector<int> dp(n+1);
    dp[0] = 0;
    dp[1] = 1;
    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i-1] + dp[i-2];
    }
    return dp[n];
}

int main() {
    int n = 10;
    cout << fib(n) << endl;  // 10번째 피보나치 수 출력
    return 0;
}
```
